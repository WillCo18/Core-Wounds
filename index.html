        function generateQuiz() {

            quizSection.innerHTML = '';

            

            // Create questions

            questions.forEach((q, index) => {

                const questionDiv = document.createElement('div');

                questionDiv.classList.add('question');

                questionDiv.id = `question-${index}`;

                

                const questionTitle = document.createElement('h2');

                questionTitle.textContent = `Question ${index + 1}: ${q.text}`;

                

                const optionsDiv = document.createElement('div');

                optionsDiv.classList.add('options');

                

                // Create options

                q.options.forEach((option, optIndex) => {

                    const optionDiv = document.createElement('div');

                    optionDiv.classList.add('option');

                    optionDiv.dataset.wound = option.wound;

                    optionDiv.textContent = option.text;

                    optionDiv.addEventListener('click', () => selectOption(index, optIndex, option.wound));

                    

                    optionsDiv.appendChild(optionDiv);

                });

                

                questionDiv.appendChild(questionTitle);

                questionDiv.appendChild(optionsDiv);

                quizSection.appendChild(questionDiv);

            });

            const buttonsDiv = document.createElement('div');

        function selectOption(questionIndex, optionIndex, wound) {

            // Clear previous selection

            const allOptions = document.querySelectorAll(`#question-${questionIndex} .option`);

            allOptions.forEach(opt => opt.classList.remove('selected'));

            

            // Select current option

            allOptions[optionIndex].classList.add('selected');

            

            // Store the answer

            userAnswers[questionIndex] = { optionIndex, wound };

            

            // Enable next button

            document.getElementById('next-btn').disabled = false;

        }

            });

            

            // Show current question

            const currentQ = document.getElementById(`question-${index}`);

            currentQ.style.display = 'block';

            currentQ.classList.add('active');

            

            // Update buttons

            const prevButton = document.getElementById('prev-btn');

            const nextButton = document.getElementById('next-btn');

            

            prevButton.disabled = index === 0;

            

            if (index === questions.length - 1) {

                nextButton.textContent = 'See Results';

            } else {

                nextButton.textContent = 'Next';

            }

            

            // Disable next button if no option selected

            nextButton.disabled = !userAnswers[index];

            

            // If previously answered, show selection

            if (userAnswers[index]) {

                const selectedOption = document.querySelectorAll(`#question-${index} .option`)[userAnswers[index].optionIndex];

                selectedOption.classList.add('selected');

            }

            

            currentQuestion = index;

            updateProgressBar();

        }


        function prevQuestion() {

            if (currentQuestion > 0) {

                showQuestion(currentQuestion - 1);

            }

        }


        function nextQuestion() {

            if (currentQuestion < questions.length - 1) {

                showQuestion(currentQuestion + 1);

            } else {

                calculateResults();

            }

        }


        function updateProgressBar() {

            const progress = ((currentQuestion + 1) / questions.length) * 100;

            progressBar.style.width = `${progress}%`;

        }


        function calculateResults() {

            // Reset scores

            scores = { notEnough: 0, unworthy: 0, powerless: 0, alone: 0 };

            

            // Calculate scores from answers

            userAnswers.forEach(answer => {

                scores[answer.wound]++;

            });

            

            // Find primary and secondary wounds

            let wounds = Object.entries(scores)

                .sort((a, b) => b[1] - a[1])

                .map(item => ({ type: item[0], score: item[1] }));

            

            const primaryWound = wounds[0];

            const secondaryWound = wounds[1];

            

            // Check if there's a tie

            const isTie = primaryWound.score === secondaryWound.score;

            

            // Display results

            displayResults(primaryWound, secondaryWound, isTie);

        }


        function displayResults(primary, secondary, isTie) {

            quizSection.style.display = 'none';

            resultsSection.style.display = 'block';

            

            // Display primary result

            const primaryResultDiv = document.getElementById('primary-result');

            primaryResultDiv.innerHTML = `

                <h3>🔍 Your ${isTie ? 'First' : 'Primary'} Core Wound: "${woundDescriptions[primary.type].title}"</h3>

                <p>${woundDescriptions[primary.type].description}</p>

            `;

            

            // Display secondary result

            const secondaryResultDiv = document.getElementById('secondary-result');

            

            if (primary.score - secondary.score <= 2) {

                // There's a significant secondary wound

                secondaryResultDiv.style.display = 'block';

                secondaryResultDiv.innerHTML = `

                    <h3>🌿 Your ${isTie ? 'Second' : 'Secondary'} Core Wound: "${woundDescriptions[secondary.type].title}"</h3>

                    <p>${woundDescriptions[secondary.type].description}</p>

                `;

            } else {

                // No significant secondary wound

                secondaryResultDiv.style.display = 'none';

            }

            

            // Display limited strategies as a preview

            const strategiesDiv = document.getElementById('strategies-content');

            

            let strategiesHTML = '<ul>';

            

            // Show only ONE primary wound strategy as a teaser

            strategiesHTML += `<li>${woundDescriptions[primary.type].strategies[0]}</li>`;

            

            // Add ONE secondary wound strategy if significant

            if (primary.score - secondary.score <= 2) {

                strategiesHTML += `<li>${woundDescriptions[secondary.type].strategies[0]}</li>`;

            }

            

            strategiesHTML += '</ul>';

            strategiesDiv.innerHTML = strategiesHTML;

        }
